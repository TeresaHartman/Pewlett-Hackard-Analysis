-- Use Dictinct with Orderby to remove duplicate rows
SELECT DISTINCT ON (e.emp_no)
e.emp_no,
e.first_name,
e.last_name, 
t.title
--t.from_date,
--t.to_date
INTO unique_titles
FROM employees as e 
INNER JOIN titles as t
ON e.emp_no = t.emp_no
WHERE (e.birth_date BETWEEN '1952.01.01' AND '1955-12-31') 
      AND (t.to_date = '9999-01-01')
ORDER BY e.emp_no;
